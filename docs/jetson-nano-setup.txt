Jetpack 4.3 auf SD-Karte flashen https://developer.nvidia.com/embedded/learn/get-started-jetson-nano-devkit#write

OpenCV builden und installieren
	git clone https://github.com/mdegans/nano_build_opencv.git
	cd nano_build_opencv
	./build_opencv.sh

(Optional) .Net core SDK installieren
	wget https://download.visualstudio.microsoft.com/download/pr/781cb53b-046c-45fb-b18e-97ad65ff61a0/5c6ce7f4e031dad7cca0fdd5bcf4335b/dotnet-sdk-3.1.200-linux-arm64.tar.gz
	mkdir -p $HOME/dotnet && tar zxf dotnet-sdk-3.1.200-linux-arm64.tar.gz -C $HOME/dotnet
	export DOTNET_ROOT=$HOME/dotnet
	export PATH=$PATH:$HOME/dotnet
	
Darknet builden und installieren
	git clone https://github.com/AlexeyAB/darknet.git
	cd darknet
	vim Makefile
		GPU=1
		CUDNN=1
		OPENCV=1
		LIBSO=1
		nvcc path = /usr/local/cuda/bin/nvcc
	make	(ggf. make clean && make)
